00030                                  "abcdefghijklmnopqrstuvwxyz";

00033     construct = "_Construct";
00034   size_t lenconstruct = construct.length();

00036     end_offset = s.find_first_not_of(words, offset+lenconstruct);
00037     if (words.find(s[offset-1]) != string::npos || offset+lenconstruct == end_offset) {
00038       // make sure construct starts the word
00039       offset += lenconstruct;
00040       continue;
00041     }
00042     archs.insert(string(s, offset+lenconstruct, end_offset - offset - lenconstruct));
